import nltk from nltk.corpus import wordnet nltk.download('wordnet') def get_meaning(): query = input("Enter word: ") synsets = wordnet.synsets(query) if synsets: for synset in synsets: print(f"{synset.pos()}: {', '.join(synset.definition().split(';'))}") else: print("Meaning not found for the given word.") def main(): while True: get_meaning() if __name__ == "__main__": main()
